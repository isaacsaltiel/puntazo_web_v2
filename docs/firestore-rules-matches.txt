// y NO redefine ninguna otra colección. Cuando se despliegue, debe
// insertarse dentro del `service cloud.firestore { match /databases/...
// /documents { ... } }` ya existente, junto a las reglas actuales.
//
// Diseño general (ver docs/matches-schema.md para schema completo):
//
//   READ:
//     - dueño (resource.data.userId == auth.uid): siempre
//     - cualquiera (incl. anónimo): solo si status == "ended"
//       (motivo: el resumen estilo Strava se comparte públicamente)
//
//   CREATE:
//     - request.auth != null
//     - userId == auth.uid (no se puede crear partido a nombre de otro)
//     - status == "active"
//     - startedAt y createdAt deben ser serverTimestamp (== request.time)
//
//   UPDATE:
//     - solo el dueño (resource.data.userId == auth.uid)
//     - inmutables: userId, loc, can, lado, startedAt, createdAt
//     - status válido: "active" | "ended" | "cancelled"
//     - endedAt: settable UNA SOLA VEZ
//         · si resource.data.endedAt es null  → cualquier valor OK
//         · si resource.data.endedAt != null → debe quedar igual
//
//   DELETE:
//     - denegado. Para "borrar" se usa status = "cancelled" vía update.
//
// -----------------------------------------------------------------
// BLOQUE A AGREGAR DENTRO DE `match /databases/{database}/documents`:
// -----------------------------------------------------------------

match /matches/{matchId} {

  function statusValid(s) {
    return s == "active" || s == "ended" || s == "cancelled";
  }

  function fieldUnchanged(name) {
    return request.resource.data[name] == resource.data[name];
  }

  // READ — dueño siempre; otros solo si el partido ya terminó.
  allow read: if (request.auth != null && request.auth.uid == resource.data.userId)
              || (resource.data.status == "ended");

  // CREATE — usuario autenticado, no puede falsificar userId,
  // status inicial debe ser "active", timestamps validados con request.time.
  allow create: if request.auth != null
                && request.resource.data.userId == request.auth.uid
                && request.resource.data.status == "active"
                && request.resource.data.startedAt == request.time
                && request.resource.data.createdAt == request.time
                && statusValid(request.resource.data.status);

  // UPDATE — solo el dueño; campos inmutables protegidos; endedAt write-once.
  allow update: if request.auth != null
                && request.auth.uid == resource.data.userId
                && fieldUnchanged('userId')
                && fieldUnchanged('loc')
                && fieldUnchanged('can')
                && fieldUnchanged('lado')
                && fieldUnchanged('startedAt')
                && fieldUnchanged('createdAt')
                && statusValid(request.resource.data.status)
                && (
                     resource.data.endedAt == null
                     || request.resource.data.endedAt == resource.data.endedAt
                   );

  // DELETE — denegado. Borrado lógico via status = "cancelled".
  allow delete: if false;
}
